/**
 * Represents a family.
 * @class
 * @category Models
 * @property {number} id - The family ID.
 * @property {string} description - The family description.
 * @property {string} tcd - ???.
 * @property {string} creation_date - The family creation date.
 * @property {string} last_update - The family last update date.
 * @property {string} odid - The odid of the last {@link Customer} who modified the family.
 * @property {number} total_vat - The family total {@link VAT}.
 */

export class Family {

    /**
     * @constructor
     * @param {Object} family - The family data as an object just like it is in the database.
     * @description - This constructor is used to convert a family object into the format required by the API.
     */
    constructor(family) {
        this.id = family.FAID;
        this.description = family.FADESC;
        this.tcd = family.FATCD; // ???
        this.creation_date = family.FACREA;
        this.last_update = family.FAMOD;
        this.odid = family.FAODID; 
        this.total_vat = family.FATOTVAT;
    }

    /**
     * @method
     * @param {Family} family - The family data.
     * @returns {Object} - The family data in the format required by the API.
     * @description - This method is used to convert a family object into the format required by the API.
     */
    static reverse(family) {
        return {
            FAID : family.id,
            FADESC : family.description,
            FATCD : family.tcd,
            FACREA : family.creation_date,
            FAMOD : family.last_update,
            FAODID : family.odid,
            FATOTVAT : family.total_vat
        }
    }
}

    }
}
